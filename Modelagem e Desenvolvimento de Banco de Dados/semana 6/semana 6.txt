Banco de dados

CREATE TABLE Salas (
SalaID INT PRIMARY KEY,
NomeSala VARCHAR(50),
Capacidade INT
);

CREATE TABLE Reservas (
ReservaID INT PRIMARY KEY AUTO_INCREMENT,
SalaID INT,
DataReserva DATETIME,
Usuario VARCHAR(100),
FOREIGN KEY (SalaID) REFERENCES Salas(SalaID)
);

Transações simples

BEGIN TRANSACTION;
IF NOT EXISTS (
SELECT * FROM Reservas WHERE SalaID=1 AND DataReserva='2024-08-21 10:00:00'
)
BEGIN
INSERT INTO Reservas (SalaID, DataReserva, Usuario) VALUES (1,'2024-08-21 10:00:00','usuario');
COMMIT;
END
ELSE
BEGIN
ROLLBACK;
END;

BEGIN TRANSACTION;

END TRY
BEGIN CATCH
